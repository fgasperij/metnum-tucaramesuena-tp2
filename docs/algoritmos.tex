\subsection{Algoritmos}

% MÉTODO DE LA POTENCIA
\begin{algorithm}
\caption{metodoDeLaPotencia(Matriz $A$, Vector $v$, Int $iter$)}
\label{pseudo:metodoDeLaPotencia}
\begin{algorithmic}
  \FOR {$i=1$ hasta $iter$}
    \STATE $y = Ax$
    \STATE $x = \frac{y}{\parallel y \parallel}$
  \ENDFOR
\end{algorithmic}
\end{algorithm}
